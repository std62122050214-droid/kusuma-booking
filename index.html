        // Save booking data

        function saveBookingData() {

        // Get room status

        function getRoomStatus(roomId) {

            const bookings = bookingData[roomId] || [];

            const room = findRoomById(roomId);

           

            if (bookings.length === 0) return 'available';

            if (bookings.length < room.capacity) return 'partial';

            return 'full';

        }

 

        // Find room by ID

        function findRoomById(roomId) {

            for (const gender in roomConfig) {

                for (const building in roomConfig[gender]) {

                    const room = roomConfig[gender][building].find(r => r.id === roomId);

                    if (room) return room;

                }

            }

            return null;

        }

        // Submit booking

        function submitBooking(event, roomId) {

            event.preventDefault();

           

            const employeeId = document.getElementById('employeeId').value.trim();

            const guestName = document.getElementById('guestName').value.trim();

 

            if (!employeeId || !guestName) {

                alert('กรุณากรอกข้อมูลให้ครบถ้วน');

                return;

            }

 

            // Check if employee already booked

            for (const roomKey in bookingData) {

                const existingBooking = (bookingData[roomKey] || []).find(b => b.employeeId === employeeId);

                if (existingBooking) {

                    alert(`รหัสพนักงาน ${employeeId} ได้จองห้องแล้วในห้อง ${findRoomById(roomKey).name}`);

                    return;

                }

            }

 

            if (!bookingData[roomId]) {

                bookingData[roomId] = [];

            }

 

            const room = findRoomById(roomId);

            if (bookingData[roomId].length >= room.capacity) {

                alert('ห้องเต็มแล้ว');

                return;

            }

 

            bookingData[roomId].push({

                employeeId: employeeId,

                name: guestName,

                timestamp: new Date().toISOString()

            });

 

            saveBookingData();

            renderAllRooms();

            closeModal();

           

            alert(`จองห้อง ${room.name} สำเร็จ!`);

        }

 

        // Remove booking

        function removeBooking(roomId, index) {

            if (confirm('ต้องการยกเลิกการจองนี้หรือไม่?')) {

                bookingData[roomId].splice(index, 1);

                if (bookingData[roomId].length === 0) {

                    delete bookingData[roomId];

                }

                saveBookingData();

                renderAllRooms();

                closeModal();

                handleRoomClick(roomId); // Re-open modal to show updated list

            }

        }

            modal.classList.add('hidden');

            modal.classList.remove('flex');

        }

                closeModal();

            }

        });
